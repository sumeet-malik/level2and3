
class Solution
{
    public boolean isEularCircuitExist(int V, ArrayList<ArrayList<Integer>> adj)
    {
        // Code here
        int count = 0;
        for(int i = 0; i < V; i++){
            if(adj.get(i).size() % 2 == 1){
                count++;
            }
        }
        
        return count == 0;
    }
}
